docker-compose部署kafka+zookeeper项目

2023/5/24 项目部署 
version: '3.3'
services:
  web:
    image: nginx
    ports:
      - '8905:80'
      - '8906:81'
    links:
      - artifacts-zjaudit-kafka-consumer
      - artifacts-zjaudit-service
      - artifacts-zjaudit-startflow
      - artifacts-zjaudit-web-start
      - artifacts-ed-zjaudit-kafka-consumer
      - artifacts-ed-zjaudit-service
      - artifacts-ed-zjaudit-web-start
    volumes:
      - ./artifacts/dist:/usr/share/nginx/html/
      - ./artifacts-ed/dist:/usr/share/nginx/html-ab/
      - ./nginx.conf:/etc/nginx/nginx.conf

  artifacts-zjaudit-kafka-consumer:
    privileged: true
    image: java8base:v1
    ports:
      - "20201:20201"
    working_dir: /root/artifacts/zjaudit-kafka-consumer
    volumes:
      - ./artifacts/zjaudit-kafka-consumer:/root/artifacts/zjaudit-kafka-consumer
    command: java -Djava.compiler=NONE org.springframework.boot.loader.JarLauncher

  artifacts-zjaudit-service:
    privileged: true
    image: java8base:v1
    working_dir: /root/artifacts/zjaudit-service
    volumes:
      - ./artifacts/zjaudit-service:/root/artifacts/zjaudit-service
    command: java -Djava.compiler=NONE org.springframework.boot.loader.JarLauncher

  artifacts-zjaudit-startflow:
    privileged: true
    image: java8base:v1
    working_dir: /root/artifacts/zjaudit-startflow
    volumes:
      - ./artifacts/zjaudit-startflow:/root/artifacts/zjaudit-startflow
      - ./r2Workspace/oracle/ResourceOneHome_shentong:/data/zjaudit/r1Workspace/oracle/ResourceOneHome_shentong
    command: java -Djava.compiler=NONE org.springframework.boot.loader.JarLauncher


  artifacts-zjaudit-web-start:
    privileged: true
    image: java8base:v1
    working_dir: /root/artifacts/zjaudit-web-start  
    volumes:
      - ./artifacts/zjaudit-web-start:/root/artifacts/zjaudit-web-start
      - ./r1Workspace/oracle/ResourceOneHome_shentong:/data/zjaudit/r1Workspace/oracle/ResourceOneHome_shentong
    command: java -Djava.compiler=NONE org.springframework.boot.loader.JarLauncher


  artifacts-ed-zjaudit-kafka-consumer:
    privileged: true
    image: java8base:v1
    working_dir: /root/artifacts-ed/zjaudit-kafka-consumer
    volumes:
      - ./artifacts-ed/zjaudit-kafka-consumer:/root/artifacts-ed/zjaudit-kafka-consumer
    command: java -Djava.compiler=NONE org.springframework.boot.loader.JarLauncher

  artifacts-ed-zjaudit-service:
    privileged: true
    image: java8base:v1
    working_dir: /root/artifacts-ed/zjaudit-service
    volumes:
      - ./artifacts-ed/zjaudit-service:/root/artifacts-ed/zjaudit-service
    command: java -Djava.compiler=NONE org.springframework.boot.loader.JarLauncher

  artifacts-ed-zjaudit-web-start:
    privileged: true
    image: java8base:v1
    working_dir: /root/artifacts-ed/zjaudit-web-start
    volumes:
      - ./artifacts-ed/zjaudit-web-start:/root/artifacts-ed/zjaudit-web-start
    command: java -Djava.compiler=NONE org.springframework.boot.loader.JarLauncher

  zookeeper:
    image: 'bitnami/zookeeper:latest'
    restart: always
    environment:
      ALLOW_ANONYMOUS_LOGIN: "yes"
      ZOO_MY_ID: 1
    ports:
      - "2181:2181"
    container_name: 'zookeeper_ZJ'
    networks:
      kafka:
        aliases:
          - zookeeper

  redis:
    image: redis
    privileged: true
    ports:
      - "8907:6379"
    environment:
      TZ: Asia/Shanghai
    volumes:
      - ./redis/conf/redis.conf:/etc/redis/redis.conf
    command: ["redis-server","/etc/redis/redis.conf"]

  kafka:
    image: bitnami/kafka:2.8.0
    ports:
      - "9092:9092"
    environment:
      - SSL=//10.10.23.88:9092
      - KAFKA_ADVERTISED_HOST_NAME=10.10.23.88
      - KAFKA_BROKER_ID=0
      - KAFKA_LISTENERS=PLAINTEXT://:9092
      - KAFKA_ADVERTISED_LISTENERS=PLAINTEXT://10.10.23.88:9092
      - KAFKA_ADVERTISED_PORT=9092
      - KAFKA_ZOOKEEPER_CONNECT=10.10.23.88:2181
      - ALLOW_PLAINTEXT_LISTENER=yes
    volumes:
#      - ./kafka/etc/localtime:/etc/localtime
      - ./kafka/kafka-logs:/kafka
    depends_on:
      - zookeeper
    links:
      - zookeeper
    networks:
      kafka:
        aliases:
          - kafka
#需要把zookeeper和Kafka放在同一网络中
networks:
  kafka:
    driver: bridge
